### 1.Install

```sh
npm install peach-design -S
```

### 2. Usage


```javascript
import React from "react";
import {Input,Button,Select} from 'peach-design';
function App() {
    return (
        <div>
            <Input/>
            <Button/>
            <Select/>
        </div>
    );
}

export default App;

```
